Criação do banco de dados:

CREATE TABLE `Artists` (
    	`Id` int NOT NULL AUTO_INCREMENT,
    	`Name` longtext CHARACTER SET utf8mb4 NOT NULL,
    	`Description` longtext CHARACTER SET utf8mb4 NOT NULL,
   	 `BirthDate` date NOT NULL,
    	CONSTRAINT `PK_Artists` PRIMARY KEY (`Id`)
	) CHARACTER SET=utf8mb4;

CREATE TABLE `MusicalGenres` (
    `Id` int NOT NULL AUTO_INCREMENT,
    `Name` longtext CHARACTER SET utf8mb4 NOT NULL,
    `Description` longtext CHARACTER SET utf8mb4 NOT NULL,
    CONSTRAINT `PK_MusicalGenres` PRIMARY KEY (`Id`)
) CHARACTER SET=utf8mb4;

CREATE TABLE `ArtistMusicalGenre` (
    `ArtistsId` int NOT NULL,
    `MusicalGenresId` int NOT NULL,
    CONSTRAINT `PK_ArtistMusicalGenre` PRIMARY KEY (`ArtistsId`, `MusicalGenresId`),
    CONSTRAINT `FK_ArtistMusicalGenre_Artists_ArtistsId` FOREIGN KEY (`ArtistsId`) REFERENCES `Artists` (`Id`) ON DELETE CASCADE,
    CONSTRAINT `FK_ArtistMusicalGenre_MusicalGenres_MusicalGenresId` FOREIGN KEY (`MusicalGenresId`) REFERENCES `MusicalGenres` (`Id`) ON DELETE CASCADE
) CHARACTER SET=utf8mb4;
