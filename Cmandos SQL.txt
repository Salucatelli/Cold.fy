Criação do banco de dados:

CREATE TABLE `Artists` (
    	`Id` int NOT NULL AUTO_INCREMENT,
    	`Name` longtext CHARACTER SET utf8mb4 NOT NULL,
    	`Description` longtext CHARACTER SET utf8mb4 NOT NULL,
   	 `BirthDate` date NOT NULL,
    	CONSTRAINT `PK_Artists` PRIMARY KEY (`Id`)
	) CHARACTER SET=utf8mb4;

CREATE TABLE `MusicalGenres` (
    `Id` int NOT NULL AUTO_INCREMENT,
    `Name` longtext CHARACTER SET utf8mb4 NOT NULL,
    `Description` longtext CHARACTER SET utf8mb4 NOT NULL,
    CONSTRAINT `PK_MusicalGenres` PRIMARY KEY (`Id`)
) CHARACTER SET=utf8mb4;

CREATE TABLE `ArtistMusicalGenre` (
    `ArtistsId` int NOT NULL,
    `MusicalGenresId` int NOT NULL,
    CONSTRAINT `PK_ArtistMusicalGenre` PRIMARY KEY (`ArtistsId`, `MusicalGenresId`),
    CONSTRAINT `FK_ArtistMusicalGenre_Artists_ArtistsId` FOREIGN KEY (`ArtistsId`) REFERENCES `Artists` (`Id`) ON DELETE CASCADE,
    CONSTRAINT `FK_ArtistMusicalGenre_MusicalGenres_MusicalGenresId` FOREIGN KEY (`MusicalGenresId`) REFERENCES `MusicalGenres` (`Id`) ON DELETE CASCADE
) CHARACTER SET=utf8mb4;


CREATE TABLE `Album` (
    `Id` int NOT NULL AUTO_INCREMENT,
    `Name` longtext CHARACTER SET utf8mb4 NOT NULL,
    `Description` longtext CHARACTER SET utf8mb4 NOT NULL,
    `ReleaseDate` date NOT NULL,
    `ArtistId` int NOT NULL,
    CONSTRAINT `PK_Album` PRIMARY KEY (`Id`),
    CONSTRAINT `FK_Album_Artists_ArtistId` FOREIGN KEY (`ArtistId`) REFERENCES `Artists` (`Id`) ON DELETE CASCADE
) CHARACTER SET=utf8mb4;


CREATE TABLE `Music` (
    `Id` int NOT NULL AUTO_INCREMENT,
    `Name` longtext CHARACTER SET utf8mb4 NOT NULL,
    `Duration` int NOT NULL,
    `ArtistId` int NOT NULL,
    CONSTRAINT `PK_Music` PRIMARY KEY (`Id`),
    CONSTRAINT `FK_Music_Artists_ArtistId` FOREIGN KEY (`ArtistId`) REFERENCES `Artists` (`Id`) ON DELETE CASCADE
) CHARACTER SET=utf8mb4;


CREATE TABLE `Playlist` (
    `Id` int NOT NULL AUTO_INCREMENT,
    `Name` longtext CHARACTER SET utf8mb4 NOT NULL,
    `Code` int NOT NULL,
    `UserId` varchar(255) CHARACTER SET utf8mb4 NOT NULL,
    CONSTRAINT `PK_Playlist` PRIMARY KEY (`Id`),
    CONSTRAINT `FK_Playlist_AspNetUsers_UserId` FOREIGN KEY (`UserId`) REFERENCES `AspNetUsers` (`Id`) ON DELETE CASCADE
) CHARACTER SET=utf8mb4;


CREATE TABLE `AlbumMusic` (
    `AlbumsId` int NOT NULL,
    `MusicsId` int NOT NULL,
    CONSTRAINT `PK_AlbumMusic` PRIMARY KEY (`AlbumsId`, `MusicsId`),
    CONSTRAINT `FK_AlbumMusic_Album_AlbumsId` FOREIGN KEY (`AlbumsId`) REFERENCES `Album` (`Id`) ON DELETE CASCADE,
    CONSTRAINT `FK_AlbumMusic_Music_MusicsId` FOREIGN KEY (`MusicsId`) REFERENCES `Music` (`Id`) ON DELETE CASCADE
) CHARACTER SET=utf8mb4;


CREATE TABLE `MusicPlaylist` (
    `MusicsId` int NOT NULL,
    `PlaylistsId` int NOT NULL,
    CONSTRAINT `PK_MusicPlaylist` PRIMARY KEY (`MusicsId`, `PlaylistsId`),
    CONSTRAINT `FK_MusicPlaylist_Music_MusicsId` FOREIGN KEY (`MusicsId`) REFERENCES `Music` (`Id`) ON DELETE CASCADE,
    CONSTRAINT `FK_MusicPlaylist_Playlist_PlaylistsId` FOREIGN KEY (`PlaylistsId`) REFERENCES `Playlist` (`Id`) ON DELETE CASCADE
) CHARACTER SET=utf8mb4;
